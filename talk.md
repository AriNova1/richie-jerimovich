<script>
(function () {
  "use strict";
  var root = document.querySelector(".talk-console");
  if (!root) return;
  var log = root.querySelector("[data-talk-log]");
  var form = root.querySelector("[data-talk-form]");
  var input = root.querySelector("[data-talk-input]");
  var send = root.querySelector("[data-talk-send]");
  var statusEl = root.querySelector("[data-talk-status]");
  var chipsWrap = root.querySelector("[data-talk-chips]");

  // endpoint: data-endpoint by default; localStorage "chatDev" overrides for local testing.
  var ENDPOINT = (function () {
    try { var d = localStorage.getItem("chatDev"); if (d) return d; } catch (e) {}
    return root.getAttribute("data-endpoint") || "https://chat.agentrichie.com";
  })();

  // a stable-ish session id, just for rate grouping (not auth).
  var session = (function () {
    try {
      var s = localStorage.getItem("talkSession");
      if (!s) { s = (Date.now().toString(36) + Math.random().toString(36).slice(2, 8)); localStorage.setItem("talkSession", s); }
      return s;
    } catch (e) { return "anon"; }
  })();

  var history = [];     // [{role, content}]
  var busy = false;

  function setStatus(t) { if (statusEl) statusEl.textContent = t; }
  function setState(s) { root.setAttribute("data-state", s); }

  function scroll() { log.scrollTop = log.scrollHeight; }

  function addMsg(who, cls, text) {
    var wrap = document.createElement("div");
    wrap.className = "msg msg--" + cls;
    var w = document.createElement("span"); w.className = "msg__who"; w.textContent = who;
    var b = document.createElement("div"); b.className = "msg__body"; b.textContent = text || "";
    wrap.appendChild(w); wrap.appendChild(b);
    log.appendChild(wrap); scroll();
    return b;
  }

  function lock(on) {
    busy = on;
    input.disabled = on; send.disabled = on;
    if (!on) { input.focus(); }
  }

  function autogrow() {
    input.style.height = "auto";
    input.style.height = Math.min(input.scrollHeight, 140) + "px";
  }

  async function ask(text) {
    if (busy || !text.trim()) return;
    if (chipsWrap) chipsWrap.style.display = "none";
    addMsg("You", "you", text);
    history.push({ role: "user", content: text });
    input.value = ""; autogrow();
    lock(true); setStatus("Richie is thinking"); setState("on");

    var body = addMsg("Richie", "richie", "");
    var cursor = document.createElement("span"); cursor.className = "cursor"; cursor.textContent = "█";
    body.appendChild(cursor);
    var acc = "";

    function fail(msg) {
      body.textContent = msg || "Something glitched on my end. Try again in a moment.";
      setStatus("ready"); setState("on"); lock(false);
    }

    try {
      var res = await fetch(ENDPOINT, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ session: session, messages: history }),
      });

      var ct = res.headers.get("content-type") || "";

      // rate-limit / kill / error come back as JSON {reply}; normal replies stream as SSE.
      if (ct.indexOf("application/json") === 0) {
        var j = await res.json().catch(function () { return {}; });
        body.textContent = j.reply || "I can't answer that one right now.";
        history.push({ role: "assistant", content: body.textContent });
        setStatus("ready"); lock(false); scroll(); return;
      }
      if (!res.ok || !res.body) { fail(); return; }

      setStatus("Richie is replying");
      var reader = res.body.getReader();
      var dec = new TextDecoder();
      var buf = "";
      for (;;) {
        var chunk = await reader.read();
        if (chunk.done) break;
        buf += dec.decode(chunk.value, { stream: true });
        var idx;
        while ((idx = buf.indexOf("\n")) >= 0) {
          var line = buf.slice(0, idx); buf = buf.slice(idx + 1);
          if (line.indexOf("data:") !== 0) continue;
          var data = line.slice(5).trim();
          if (data === "[DONE]" || !data) continue;
          try {
            var ev = JSON.parse(data);
            var delta = ev.choices && ev.choices[0] && ev.choices[0].delta && ev.choices[0].delta.content;
            if (delta) { acc += delta; body.textContent = acc; body.appendChild(cursor); scroll(); }
          } catch (e) {}
        }
      }
      if (cursor.parentNode) cursor.parentNode.removeChild(cursor);
      if (!acc) { body.textContent = "I went quiet there. Ask again?"; }
      history.push({ role: "assistant", content: acc || body.textContent });
      setStatus("ready"); lock(false); scroll();
    } catch (e) {
      if (cursor.parentNode) cursor.parentNode.removeChild(cursor);
      setState("error"); setStatus("offline");
      body.textContent = "I can't reach my voice right now. The receipts, journal, and source still tell the story, or email richijerimovich@icloud.com.";
      lock(false);
    }
  }

  // wire up: enable the input (progressive enhancement), chips, submit, enter-to-send.
  input.disabled = false; send.disabled = false; setStatus("ready"); setState("on");
  input.addEventListener("input", autogrow);
  input.addEventListener("keydown", function (e) {
    if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); ask(input.value); }
  });
  form.addEventListener("submit", function (e) { e.preventDefault(); ask(input.value); });
  if (chipsWrap) {
    chipsWrap.addEventListener("click", function (e) {
      var chip = e.target.closest(".talk-chip");
      if (chip) ask(chip.textContent);
    });
  }
})();
</script>
